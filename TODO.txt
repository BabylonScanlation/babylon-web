(function() {
  if (performance.setResourceTimingBufferSize) {
    performance.setResourceTimingBufferSize(500);
  }

  const resources = performance.getEntriesByType('resource');
  const header = `NAME`.padEnd(60) + ` | STATUS | DOMAIN`.padEnd(25) + ` | TYPE`.padEnd(8) + ` | SIZE`.padEnd(10) + ` | TIME`.padEnd(8) + ` | CACHED\n`;
  const divider = `-`.repeat(130) + `\n`;

  const rows = resources.map(r => {
    try {
      const url = new URL(r.name);
      
      // Extraer los últimos 2 segmentos del path para que se vea la diferencia (ej: utils/path.js vs libreria/path.js)
      const segments = url.pathname.split('/').filter(Boolean);
      let name = segments.slice(-2).join('/') || url.hostname;
      if (url.search) name += url.search;
      
      if (name.length > 57) name = name.substring(0, 25) + '...' + name.substring(name.length - 29);

      const domain = url.hostname;
      let initiator = r.initiatorType.toUpperCase();
      if (initiator === 'OTHER' && name.includes('.webp')) initiator = 'FETCH'; 

      const sizeBytes = r.encodedBodySize || r.transferSize || r.decodedBodySize || 0;
      const size = sizeBytes > 0 ? (sizeBytes / 1024).toFixed(1) + ' KB' : '(unknown)';
      const time = r.duration.toFixed(0) + 'ms';
      const isCache = (r.transferSize === 0 && r.decodedBodySize > 0) ? 'YES' : 'NO';

      return `${name.padEnd(60)} | 200    | ${domain.padEnd(22)} | ${initiator.padEnd(6)} | ${size.padEnd(8)} | ${time.padEnd(6)} | ${isCache}`;
    } catch (e) {
      return `Error: ${r.name}`;
    }
  });

  const result = header + divider + rows.join('\n');
  const el = document.createElement('textarea');
  el.value = result;
  document.body.appendChild(el);
  el.select();
  document.execCommand('copy');
  document.body.removeChild(el);
  console.log(`%c \u2705 Copiado (${rows.length} de ${resources.length} peticiones registradas). Se añadieron rutas para evitar confusiones de nombres.`, "color: #00ff00; font-weight: bold; font-size: 14px;");
})();
